GUI Functions


EXAMPLES:
// init main class
let GUI = new PXEngine_1.0();
GUI.init();
// show & close frame
GUI.show(youframeid);
GUI.close(youframeid);
// switcher frame : state - true = display: block; false = display:none;
GUI.frame(youframeid,state);
// animation load : create 4 div tags to id [bl0 ,bl1 ,bl2, bl3]
// start animation
GUI.animload();


Audio Functions


EXAMPLES:
// init main class : audio_tagid - audio tag id
GUI.initAudio(audio_tagid);
// load File
GUI.pxLoadFile(file);
GUI.AudioLoad();
// play audio
GUI.pxPlay();
// pause audio
GUI.pxPause();
